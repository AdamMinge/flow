# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Set CMake Version ----------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
cmake_minimum_required(VERSION 3.11)
# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Set files variable ---------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
set(source_root ${FLOW_SOURCE_DIR}/src/flow/utils/qtlineedit)
set(include_root ${FLOW_SOURCE_DIR}/include/flow/utils/qtlineedit)
set(resources_root ${FLOW_SOURCE_DIR}/resources)

set(sources
        ${source_root}/qtlineeditwithhistory.cpp
        ${source_root}/qtfilebrowserlineedit.cpp)

set(includes
        ${include_root}/export.h
        ${include_root}/qtlineeditwithhistory.h
        ${include_root}/qtfilebrowserlineedit.h)

set(resources
        ${resources_root}/flow-utils-images.qrc)
# -------------------------------------------------------------------------------------------------- #
# ---------------------------------- Qt libraries configuration ------------------------------------ #
# -------------------------------------------------------------------------------------------------- #
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)
# -------------------------------------------------------------------------------------------------- #
# ----------------------------------- Find external libraries -------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
find_package(Qt6 REQUIRED COMPONENTS Widgets)
# -------------------------------------------------------------------------------------------------- #
# ------------------------------------ Create utils shared lib ------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
flow_add_utils(qtlineedit
        SOURCES ${sources} ${includes} ${resources}
        DEPENDS Qt6::Widgets)
# -------------------------------------------------------------------------------------------------- #
